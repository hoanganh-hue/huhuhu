125 def validate_record(self, record: Dict[str, Any]) -> Dict[str, Any]: 

126 """ 

127 Kiểm tra và validate một record dữ liệu

129 Args: 

130 record: Record dữ liệu cần kiểm tra 

131  

132 Returns:

134 """ 

135 errors = [] 

136 warnings = [] 

137 cleaned_record = {} 

138

140 cccd = self.clean_cccd_number(record.get('cccd', ''))

458 def _names_match(self, name1: str, name2: str) -> bool: 

459 """ 

460 Kiểm tra tên có khớp nhau không (fuzzy matching) 

461  

462 Args: 

463 name1: Tên thứ nhất 

464 name2: Tên thứ hai 

465  

466 Returns: 

467 True nếu tên khớp 

468 """ 

469 if not name1 or not name2: 

470 return False 

471 

472 # Chuẩn hóa tên 

473 name1_clean = name1.lower().strip() 

474 name2_clean = name2.lower().strip() 

475 

476 # Kiểm tra exact match 

477 if name1_clean == name2_clean: 

478 return True 

479 

480 # Kiểm tra partial match (một tên chứa tên kia) 

481 if name1_clean in name2_clean or name2_clean in name1_clean: 

482 return True 

483 

484 # Kiểm tra từ khóa chính 

485 name1_words = set(name1_clean.split()) 

486 name2_words = set(name2_clean.split()) 

487 

488 # Nếu có ít nhất 2 từ khớp nhau 

489 common_words = name1_words.intersection(name2_words) 

490 if len(common_words) >= 2: 

491 return True 

492 

493 return False

495 def _calculate_completeness(self, record: Dict) -> int: 

496 """ 

497 Tính độ đầy đủ của dữ liệu 

498  

499 Args: 

500 record: Record cần tính 

501  

502 Returns: 

503 Phần trăm độ đầy đủ (0-100) 

504 """ 

505 fields = ['cccd', 'name', 'phone', 'address', 'bhxh_number', 'birth_date'] 

506 filled_fields = 0 

507 

508 for field in fields: 

509 if record.get(field) and str(record.get(field)).strip(): 

510 filled_fields += 1 

511 

512 return int((filled_fields / len(fields)) * 100)

514 def _extract_birth_date_from_cccd(self, cccd: str) -> str: 

515 """ 

516 Trích xuất ngày tháng năm sinh từ số CCCD 

517  

518 Args: 

519 cccd: Số CCCD (12 chữ số) 

520  

521 Returns: 

522 Ngày tháng năm sinh theo định dạng DD/MM/YYYY 

523 """ 

524 if not cccd or len(cccd) != 12: 

525 return "" 

526 

527 try: 

528 # CCCD có cấu trúc: 3 chữ số đầu (tỉnh) + 1 chữ số (giới tính) + 2 chữ số (năm sinh) + 6 chữ số (số thứ tự) 

529 # Trích xuất năm sinh (2 chữ số) 

530 year_part = cccd[4:6] 

531 

532 # Trích xuất giới tính để xác định thế kỷ 

533 gender_digit = int(cccd[3]) 

534 

535 # Xác định thế kỷ dựa trên giới tính 

536 if gender_digit in [0, 1, 2, 3, 4]: 

537 century = "20" # Nam: 20xx 

538 else: 

539 century = "20" # Nữ: 20xx (hoặc có thể là 19xx tùy theo logic) 

540 

541 # Tạo năm đầy đủ 

542 full_year = century + year_part 

543 

544 # Trích xuất ngày và tháng từ 6 chữ số cuối (số thứ tự) 

545 # Sử dụng 2 chữ số đầu của số thứ tự làm ngày, 2 chữ số tiếp theo làm tháng 

546 sequence = cccd[6:12] 

547 day = sequence[:2] 

548 month = sequence[2:4] 

549 

550 # Validate và format 

551 day_int = int(day) 

552 month_int = int(month) 

553 year_int = int(full_year) 

554 

555 # Kiểm tra tính hợp lệ 

556 if 1 <= day_int <= 31 and 1 <= month_int <= 12 and 1900 <= year_int <= 2024: 

557 return f"{day_int:02d}/{month_int:02d}/{year_int}" 

558 else: 

559 return "" 

560 

561 except (ValueError, IndexError): 

562 return ""

651 def save_debug_csv(self, data: List[Dict], file_path: Union[str, Path]) -> bool: 

652 """ 

653 Lưu dữ liệu debug ra file CSV với raw JSON response 

654  

655 Args: 

656 data: Dữ liệu cần lưu 

657 file_path: Đường dẫn file CSV 

658  

659 Returns: 

660 True nếu thành công 

661 """ 

662 try: 

663 import csv 

664 

665 file_path = Path(file_path) 

666 file_path.parent.mkdir(parents=True, exist_ok=True) 

667 

668 if not data: 

669 return True 

670 

671 # Lấy tất cả các key từ dữ liệu 

672 all_keys = set() 

673 for record in data: 

674 all_keys.update(record.keys()) 

675 

676 # Sắp xếp keys để có thứ tự nhất quán 

677 sorted_keys = sorted(all_keys) 

678 

679 with open(file_path, 'w', newline='', encoding='utf-8') as csvfile: 

680 writer = csv.DictWriter(csvfile, fieldnames=sorted_keys) 

681 writer.writeheader() 

682 

683 for record in data: 

684 # Chuẩn hóa dữ liệu cho CSV 

685 csv_record = {} 

686 for key in sorted_keys: 

687 value = record.get(key, '') 

688 

689 # Xử lý raw_response (JSON) 

690 if key == 'raw_response' and value: 

691 if isinstance(value, dict): 

692 csv_record[key] = json.dumps(value, ensure_ascii=False) 

693 else: 

694 csv_record[key] = str(value) 

695 else: 

696 csv_record[key] = str(value) if value is not None else '' 

697 

698 writer.writerow(csv_record) 

699 

700 return True 

701 except Exception as e: 

702 print(f"Lỗi khi lưu debug CSV: {e}") 

703 return False

848 def remove_duplicates(self, data: List[Dict], key_fields: List[str] = None) -> List[Dict]: 

849 """ 

850 Loại bỏ dữ liệu trùng lặp dựa trên các trường key 

851  

852 Args: 

853 data: Danh sách dữ liệu cần deduplication 

854 key_fields: Danh sách các trường dùng để xác định duplicate (mặc định: ['cccd']) 

855  

856 Returns: 

857 Danh sách dữ liệu đã loại bỏ duplicate 

858 """ 

859 if not data: 

860 return [] 

861 

862 if key_fields is None: 

863 key_fields = ['cccd'] 

864 

865 # Validate không có mock data 

866 try: 

867 validate_no_mock_data(data, "DataProcessor.remove_duplicates") 

868 except DataUnavailableError as e: 

869 raise DataUnavailableError(f"Deduplication failed: {e}") 

870 

871 # Tạo set để track các key đã gặp 

872 seen_keys = set() 

873 unique_data = [] 

874 duplicate_count = 0 

875 

876 for record in data: 

877 # Tạo key từ các trường được chỉ định 

878 key_parts = [] 

879 for field in key_fields: 

880 value = record.get(field, '') 

881 key_parts.append(str(value)) 

882 

883 record_key = '|'.join(key_parts) 

884 

885 if record_key not in seen_keys: 

886 seen_keys.add(record_key) 

887 unique_data.append(record) 

888 else: 

889 duplicate_count += 1 

890 

891 print(f"📊 Deduplication completed: {len(data)} -> {len(unique_data)} records ({duplicate_count} duplicates removed)") 

892 

893 return unique_data

895 def validate_data_integrity(self, data: List[Dict]) -> Dict[str, Any]: 

896 """ 

897 Validate tính toàn vẹn của dữ liệu 

898  

899 Args: 

900 data: Danh sách dữ liệu cần validate 

901  

902 Returns: 

903 Dict chứa kết quả validation 

904 """ 

905 validation_result = { 

906 'valid': True, 

907 'total_records': len(data), 

908 'valid_records': 0, 

909 'invalid_records': 0, 

910 'errors': [], 

911 'warnings': [] 

912 } 

913 

914 if not data: 

915 validation_result['valid'] = False 

916 validation_result['errors'].append("Data is empty") 

917 return validation_result 

918 

919 # Validate không có mock data 

920 try: 

921 validate_no_mock_data(data, "DataProcessor.validate_data_integrity") 

922 except DataUnavailableError as e: 

923 validation_result['valid'] = False 

924 validation_result['errors'].append(f"Mock data detected: {e}") 

925 return validation_result 

926 

927 # Validate từng record 

928 for i, record in enumerate(data): 

929 record_errors = [] 

930 

931 # Kiểm tra CCCD 

932 cccd = record.get('cccd', '') 

933 if not cccd: 

934 record_errors.append("Missing CCCD") 

935 elif not self.cccd_pattern.match(str(cccd)): 

936 record_errors.append(f"Invalid CCCD format: {cccd}") 

937 

938 # Kiểm tra tên 

939 name = record.get('name', '') 

940 if name and not self.name_pattern.match(str(name)): 

941 record_errors.append(f"Invalid name format: {name}") 

942 

943 # Kiểm tra số điện thoại 

944 phone = record.get('phone', '') 

945 if phone and not self.phone_pattern.match(str(phone)): 

946 record_errors.append(f"Invalid phone format: {phone}") 

947 

948 if record_errors: 

949 validation_result['invalid_records'] += 1 

950 validation_result['errors'].extend([f"Record {i+1}: {error}" for error in record_errors]) 

951 else: 

952 validation_result['valid_records'] += 1 

953 

954 # Tính tỷ lệ hợp lệ 

955 if validation_result['total_records'] > 0: 

956 valid_rate = validation_result['valid_records'] / validation_result['total_records'] 

957 if valid_rate < 0.8: # Dưới 80% hợp lệ 

958 validation_result['warnings'].append(f"Low data quality: {valid_rate:.1%} valid records") 

959 

960 validation_result['valid'] = len(validation_result['errors']) == 0 

961 

962 return validation_result
